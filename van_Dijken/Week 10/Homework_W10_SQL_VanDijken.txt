QUESTION 16

FROM BOOK
LEFT JOIN GENRE
ON GENRE=GENRE_ID
GROUP BY DESCRIPTION

QUESTION 17

FROM BOOK

QUESTION 18
SELECT AUTHOR, FIRST_NAME, LAST_NAME, COUNT(*)
FROM AUTHOR
LEFT JOIN BOOK
ON AUTHOR=AUTHOR_ID
GROUP BY AUTHOR
HAVING COUNT(*)  > 1
