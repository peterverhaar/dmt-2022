15.
SELECT DISTINCT YEAR, COUNT(TITLE)
FROM BOOK
GROUP BY YEAR;

16.
SELECT DISTINCT DESCRIPTION, AVG(SOLD) AS AVG_SOLD
FROM GENRE, BOOK
WHERE GENRE = GENRE_ID
GROUP BY DESCRIPTION;

17.
SELECT FIRST_NAME, LAST_NAME, SUM(SOLD) AS SUM_SOLD
FROM AUTHOR, BOOK
WHERE AUTHOR= AUTHOR_ID
GROUP BY AUTHOR;

18.
SELECT FIRST_NAME, LAST_NAME, COUNT(TITLE)
FROM AUTHOR, BOOK
WHERE AUTHOR=AUTHOR_ID
GROUP BY AUTHOR
HAVING COUNT(TITLE) > 1;

19.
SELECT NAME, COUNT(TITLE)
FROM PUBLISHER, BOOK
WHERE PUBLISHER=PUBL_ID
GROUP BY PUBLISHER
HAVING COUNT(TITLE) > 3;
